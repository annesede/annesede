void affineEncrypt(char *plaintext, char *ciphertext, int keyA, int keyB){
    long length = strlen(plaintext);
    for (long i = 0; i < length; ++i) {
        if (isupper(ciphertext[i])){
            ciphertext[i] = (keyA*(plaintext[i]-'A')+keyB)%26+'A';
        } else if (islower(ciphertext[i])){
            ciphertext[i] = (keyA*(plaintext[i]-'a')+keyB)%26+'a';
        } else {
            ciphertext[i] = plaintext[i];
        }
    }
}

void affineDecrypt(char *ciphertext, char *plaintext, int keyAReverse, int keyB){
    long length = strlen(ciphertext);
    for (long i = 0; i < length; ++i) {
        if (isupper(ciphertext[i])){
            plaintext[i] = ((keyAReverse*(ciphertext[i]-'A'-keyB))%26+26)%26+'A';
        } else if (islower(ciphertext[i])){
            plaintext[i] = ((keyAReverse*(ciphertext[i]-'a'-keyB))%26+26)%26+'a';
        } else {
            plaintext[i] = ciphertext[i];
        }
    }
}

void freqAnalyze(char *ciphertext, long *count){
    int num = 0;
    while (ciphertext[num] != '\0'){
        if (isupper(ciphertext[num])){
            count[ciphertext[num]-'A']++;
        } else if (islower(ciphertext[num])){
            count[ciphertext[num]-'a']++;
        }
        ++num;
    }
}

自同步序列密码

种子密钥通过LFSR(线性反馈移位寄存器)生成伪随机密序列 $k=k_0k_1k_2…$.

  • 加密 $c_i=E_{k}(m)=m_i\oplus k_i$, $\oplus$ 为异或(位加法).
  • 解密 $m_i=D_{k}(m)=c_i\oplus k_i$.

// 32-bit long
module lfsr(32)(clk, reset, lfsr);
  input clk, reset;
  output reg [31:0] lfsr;
  wire d0;

  xnor(d0, lfsr[31], lfsr[21], lfsr[1], lfsr[0]);

  always @(posedge clk, posedge reset) begin
    if(reset) begin
      lfsr <= 32'h00000001;
    end
    else begin
      lfsr <= {lfsr[30:0], d0};
    end
  end
endmodule
